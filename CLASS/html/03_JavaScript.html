DOM (Document Object Model)


        

          
HTML 웹문서, 웹페이지 문서를 객체 기반으로 표현한 것
-> HTML 문서에 작성된 내용을 tree 구조로 나타냈을 때
    각각의 태그, TEXT, COMMENT를 Node라고 한다.

Node              : document(문서) tree 구조를 구성하는 모든 개별 요소를 가르키는 기본 단위
                    웹 페이지를 구성하는 모든 것
                    -> HTML 태그, div 태그, p 태그, input 태그, 태그 내부 텍스트(글자), 주석 등
                       코드에 작성된 모든 내용은 각각의 node 라고 불린다.
                       node 내부에서 태그와 요소는 분리된다.
      
nodeJS            : 프론트엔드 코드로 작성된 모든 것에 대한 JavaScript 설정을 나타낸다.
                    nodeJS로 프론트엔드 구성이 가능한 사람  ->  JavaScript 언어 기반 프론트엔드 기술자
                    nodeJS로 백엔드 구성이 가능한 사람     ->  JavaScript 언어 기반 백엔드 기술자
                    (기술 도구는 nodeJS를 확용하지만 언어의 요구사항이 JS가 아니라 TS나 다른 언어일 수도 있다.)

요소               : element (태그와 태그 내부에 작성된 텍스트, value까지 포함하는 것을 의미)


document : {
  DOCTYPE : html, 
  HTML : {
    HEAD : {
      TITLE : {TEXT : "문서 제목"},
      STYLE : {...},
      META  : {...}, 
    },
    BODY : {
      H1      : {TEXT : "제목", ATTRIBUTE : "속성"},
      COMMENT : {TEXT : "주석 내용"},
      DIV     : {...},
      ul#test{childNodes : [], innerText : 123, }
    }
  }
}

시간 지연 함수        : window에서 제공하는 시간 관련 함수들은 모두 비동기로 동작한다.
      
동기   : 앞 코드가 수행 종료되어야 다음 코드를 수행한다.
비동기  : 코드 해석(작업)이 작성 순서대로 수행되지 않고, 여러 작업이 동시에 수행한다.
        -> 먼저 읽은 코드가 수행 완료될 때까지 기다리지 않고, 다음 코드를 수행한다.

[window].setTimeout(함수, 지연시간(ms))   : 지정한 함수를 한 번만 지연시간(ms) 후에 실행한다.             "~초 후 실행"
[window].setInterval(함수, 지연시간(ms))  : 지정한 함수를 주기적으로 반복해서 지연시간(ms) 간격으로 실행한다.  "~초마다 실행"
[window].clearInterval(setInterval())  : setInterval로 설정된 반복 작업을 멈춘다.                  "~초부터 실행 중단"

팝업창

        

        

          
웹페이지에서 팝업창을 생성  -> 웹페이지에 입력된 값을 팝업창으로 보내거나,
                        팝업창에서 입력된 값을 웹페이지로 가져올 수 있다.
                        ex) 주소 검색, 본인인증, 아이디/이메일 중복 확인

                        
1) 웹페이지에서 팝업창 생성하기

const popUpBtn = document.getElementById("popUpBtn");
popUpBtn.addEvent("click", () => {

  // 빈 배열 생성
  let options = [];

  // 팝업창 크기 지정
  const width = 400;
  const height = 600;
  
  // 화면 정중앙 팝업창 위치
  const left = (window.screen.width - width) / 2;
  const top = (window.screen.height - height) / 2;

  // 빈 배열에 크기, 위치 정보 대입
  const options =
  `
  width=${width},
  height=${height},
  left=${left},
  top=${top},
  `;

  // 
  window.open("팝업창.html", "_blank", options);
})


2) 웹페이지에 입력된 값을 팝업창에 가져오기

// 입력된 값을 대입할 팝업창 변수
const popUpResult = document.getElementById("popUpResult");

const result = opener.document.getElementById("result").value;

스톱워치

        

        

          
<span>00</span> : <span>00</span> . <span>00</span>   ->  00 : 00 . 00

1분      : count % 100 / 60
1초      : count % 100 % 60
1/100 초 : count % 100

let count = 0;
let currInterval;

// 시작 버튼 클릭 시 0.01초(10ms)마다 시간을 증가 
startBtn.addEventListener("click", (e) => {

  // 클릭한 버튼에 작성된 내용이 PAUSE인 경우
  if (e.target.textContent == "PAUSE"){

    // 버튼 내용을 START로 변경
    e.target.textContent = "START";

    // currInterval 정지 처리
    window.clearInterval(currInterval);

    // 정지했으니 아래 기능 사용 못하게 돌려보내기
    return;
  }

  // 클릭한 버튼에 작성된 내용이 START인 경우
  currInterval = window.setInterval(() => {

    // 1초씩 증가
    count++;

    // 0.1초(10ms)마다 화면에 시간 출력 함수
    output();
  }, 10);

  // 카운트 시작한 뒤 정지할 수 있도록 버튼 내부 PAUSE로 변경
  e.target.textContent = "PAUSE";
});


// 시간 출력 함수
function output() {
  let minute = Math.floor(count / 100 / 60);
  let second = Math.floor((count / 100) % 60);
  let ms = count % 100;

  // 0 붙여서 문자열로 변환해서 반환
  minute = attachZero(minute);
  second = attachZero(second);
  ms = attachZero(ms);

  // minute의 값과 화면에 출력된 분이 다를 경우
  if (list[0].innerText != minute) list[0].innerText = minute;

  // second의 값과 화면에 출력된 초가 다를 경우
  if (list[1].innerText != second) list[1].innerText = second;

  list[2].innerText = ms;
}

// 한 자리 숫자인 경우 앞에 0 붙여서 반환하는 함수
function attachZero(num) {
  if (num < 10) {
    // 한 자리 숫자인 경우
    return "0" + num;
  }
  return "" + num;
}

// RESET 버튼이 클릭 되었을 때
resetBtn.addEventListener("click", () => {

  // currInterval 제거
  window.clearInterval(currInterval);

  // count, 출력된 시간 모두 0으로

  output();

  // start 버튼 활성화 ( == 비활성화 안하겠다)
